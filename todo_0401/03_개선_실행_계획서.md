    return False

# 변경 후
@staticmethod
def _is_ai_related(repo_data: dict) -> bool:
    topics = repo_data.get("topics", [])
    desc = (repo_data.get("description") or "").lower()
    name = repo_data.get("name", "").lower()

    if any(t in AI_TOPICS for t in topics):
        return True
    if any(kw in desc for kw in AI_KEYWORDS):
        return True
    if any(kw in name for kw in ["ml", "ai", "neural", "llm", "gpt", "bert", "transformer"]):
        return True
    return False  # stars 조건 제거

### Task 2.1: 적응형 딜레이 (Rate Limit 효율 극대화)

**목표:** Rate Limit 여유분을 최대한 활용하되, 소진 위험 시 자동 감속

**변경 파일:**
- `backend/app/crawler/github_client.py` — `_request()` 내 딜레이 로직 교체

**상세 구현:**

```python
async def _request(self, method: str, url: str, **kwargs):
    # 기존: await asyncio.sleep(self.delay)  # 항상 2초
    # 변경: 적응형 딜레이
    adaptive_delay = self._calculate_delay()
    await asyncio.sleep(adaptive_delay)
    self.total_api_calls += 1
    # ... 기존 로직

def _calculate_delay(self) -> float:
    """Rate Limit remaining에 기반한 동적 딜레이"""
    if self.rate_limit_remaining > 1000:
        return 0.5   # 여유 충분: 빠르게
    elif self.rate_limit_remaining > 500:
        return 1.0   # 보통
    elif self.rate_limit_remaining > 100:
        return 2.0   # 주의
    elif self.rate_limit_remaining > 20:
        return 5.0   # 절약 모드
    else:
        return 10.0  # 거의 소진: 극도로 절약
```

**효과:**
- 시간당 API 호출: ~1800 → ~4000+ (remaining 충분 시)
- Rate Limit 소진 위험 없이 2배 이상 크롤링 속도 향상
- 장기 가동(72시간+)에도 안정적

### Task 2.3: DB 쓰기 최적화 (인메모리 캐시)

**목표:** 크롤링 중 반복되는 DB SELECT를 줄여 task 처리 속도 향상 + DB 부하 감소

**변경 파일:**
- `backend/app/crawler/crawler.py` — in-memory 캐시 추가

**상세 구현:**

```python
class CrawlerManager:
    def __init__(self):
        # ... 기존 +
        self._author_cache: dict[int, int] = {}   # github_id → db_id
        self._topic_cache: dict[str, int] = {}     # topic_name → db_id

    async def _save_author(self, session, data, detailed=False):
        github_id = data["id"]
        if github_id in self._author_cache and not detailed:
            return await session.get(Author, self._author_cache[github_id])
        # ... 기존 로직
        self._author_cache[github_id] = author.id
        return author

    async def _save_repo(self, session, data):
        # topics 처리 시 캐시 활용
        for topic_name in data.get("topics", []):
            if topic_name in self._topic_cache:
                topic_id = self._topic_cache[topic_name]
                topic = await session.get(Topic, topic_id)
            else:
                # ... 기존 조회 + 생성 로직
                self._topic_cache[topic_name] = topic.id
```

**효과:**
- Author/Topic SELECT 횟수: ~60% 감소 (같은 사용자/토픽 반복 조회 제거)
- task당 DB 호출 수 감소 → task 처리 속도 향상

## Phase 3: Author-Author 연결 정상화 + 그래프 품질 향상

> 예상 파일 변경: 2개 | 위험도: 낮음
> **핵심:** 현재 Author-Author(coworker) 연결은 3단계 구조적 결함으로 사실상 작동하지 않음.
> Phase 3은 이 결함을 순서대로 수정한다.

### Task 3.1: 기여자(Contributor) Author를 그래프 노드에 포함 + coworker SQL 수정

**목표:**
1. owner뿐 아니라 기여자도 Author 노드로 추가
2. coworker SQL의 `HAVING COUNT(*) >= 2` → `>= 1`로 완화 (같은 repo 1개만 공유해도 동료)
3. coworker SQL의 대상을 owner+기여자 전체로 확장

**변경 파일:**
- `backend/app/graph/builder.py` — `build_graph()` Step 2 + Step 6 수정

**상세 구현 (Step 2 — Author 노드 확장):**

```python
# build_graph() 내 Author 노드 수집 부분

if "author" in node_types and repo_ids:
    # (1) 기존: owner 노드 수집
    # ... 유지 ...

    # (2) 추가: 기여자 노드 (contributions >= 10인 기여자)
    top_contribs = (
        await session.execute(
            select(RepoContributor)
            .where(
                RepoContributor.repository_id.in_(repo_ids),
                RepoContributor.contributions >= 10,
            )
        )
    ).scalars().all()

    contrib_author_ids = {c.author_id for c in top_contribs} - owner_ids
    if contrib_author_ids:
        contrib_authors = (
            await session.execute(
                select(Author).where(Author.id.in_(contrib_author_ids))
            )
        ).scalars().all()
        for a in contrib_authors:
            nid = f"author:{a.id}"
            if nid not in node_ids:
                nodes.append(_author_node(a, compact))
                node_ids.add(nid)

    # author_db_ids에 기여자도 포함 (coworker SQL 대상 확대)
    author_db_ids = list(owner_ids | contrib_author_ids)
```

**상세 구현 (Step 6 — coworker SQL 완화):**

```python
# 변경 전
coworker_sql = """
    ...
    WHERE a1.author_id IN :ids AND a2.author_id IN :ids
    GROUP BY a1.author_id, a2.author_id
    HAVING COUNT(*) >= 2   -- ← 2개 이상 공유 필요
"""

# 변경 후
coworker_sql = """
    ...
    WHERE a1.author_id IN :ids AND a2.author_id IN :ids
    GROUP BY a1.author_id, a2.author_id
    HAVING COUNT(*) >= 1   -- ← 같은 repo 1개만 공유해도 연결
"""
```

**효과:**
- contributes 링크: 대부분이 렌더링되지 않던 것 → 실제 표시
- coworker 링크: `author_db_ids` 대폭 증가 + 조건 완화 → **실질적인 Author-Author 연결 생성**
- 예시: pytorch의 기여자 A, B가 같은 repo 1개만 공유해도 coworker 링크 생성

### Task 3.2: get_neighbors()에 coworker 탐색 추가

**목표:** Author 노드 클릭 시 해당 개발자와 같은 repo에 기여한 다른 Author(coworker)를 표시

**현재 문제:**
`get_neighbors()` author 케이스에서 owns, contributes 링크만 표시하고
**coworker 탐색 로직이 완전히 누락**되어 있음.

**변경 파일:**
- `backend/app/graph/builder.py` — `get_neighbors()` author 케이스 확장

**상세 구현:**

```python
elif node_type == "author":
    # 기존: owns + contributes 링크 (유지)
    # ...

    # 추가: coworker 탐색 — 같은 repo의 다른 기여자 표시
    # Step 1: 이 author가 기여한 repo 목록 수집
    contributed_repos = (
        await session.execute(
            select(RepoContributor.repository_id)
            .where(RepoContributor.author_id == db_id)
        )
    ).scalars().all()

    # Step 2: 해당 repo들의 다른 기여자 (상위 contributions 기준)
    if contributed_repos:
        coworkers = (
            await session.execute(
                select(RepoContributor.author_id, func.sum(RepoContributor.contributions).label("total"))
                .where(
                    RepoContributor.repository_id.in_(contributed_repos),
                    RepoContributor.author_id != db_id,
                )
                .group_by(RepoContributor.author_id)
                .order_by(func.sum(RepoContributor.contributions).desc())
                .limit(20)  # 상위 20명
            )
        ).all()

        for coworker_id, total_contribs in coworkers:
            coworker_author = await session.get(Author, coworker_id)
            if coworker_author:
                nid = f"author:{coworker_id}"
                if nid not in node_ids:
                    nodes.append(_author_node(coworker_author, compact=True))
                    node_ids.add(nid)
                links.append({
                    "source": f"author:{db_id}",
                    "target": nid,
                    "type": "coworker",
                    "weight": min(total_contribs, 100),
                })
```

**효과:**
- Author 클릭 → 같은 repo에서 함께 일한 동료 개발자 즉시 표시
- **그래프 탐색 시 Author→Author→Author 체인 형성** (현재 불가능 → 가능)
- 네트워크 그래프의 핵심 가치인 "개발자 간 관계"가 처음으로 시각화됨

async def _do_fetch_contributors(self, session, session_id, full_name, depth):
    count = 0
    for page in range(1, 3):  # 최대 2페이지
        data = await self.client.get_repo_contributors(full_name, per_page=30, page=page)
        if not data or not isinstance(data, list):

            select(Repository).where(Repository.full_name == full_name)
        )
        repo = result.scalar_one_or_none()
        if not repo:
            return 0

        for cd in data:
            author = await self._save_author(session, cd)
            if not author:
                continue
            # ... 기존 관계 저장 + task 생성 로직
            count += 1

        if len(data) < 30:
            break  # 더 이상 페이지 없음

    await session.commit()
    return count

# cache.py
class GraphCache:
    def invalidate_by_session(self, session_id: int | None = None):
        if session_id is None:
            self.invalidate()
            return
        prefix = f"{session_id}:"
        keys_to_delete = [k for k in self._cache if k.startswith(prefix)]
        for k in keys_to_delete:
            del self._cache[k]
        # 전체 보기(session_id=None) 캐시도 무효화
        none_keys = [k for k in self._cache if k.startswith("None:")]
        for k in none_keys:
            del self._cache[k]
```

@admin_router.get("/sessions/{session_id}/depth-stats")
async def session_depth_stats(session_id: int, request: Request, _=Depends(require_admin)):
    async with async_session_factory() as db:
        result = await db.execute(
            select(CrawlTask.depth, CrawlTask.status, func.count(CrawlTask.id))

Phase 4 (대규모 대비)      → Task 4.1 → 4.2 → 4.3
  ↓  효과: 10만+ 노드에서도 안정적 그래프 생성

Phase 5 (선택적 확장)      → 필요시
```

| Phase | 핵심 가치 | 변경 규모 |
|-------|----------|----------|
| **Phase 1** | 네트워크 확장 극대화 | 소 (1개 파일, ~20줄) |
| **Phase 2** | 서버 부담 없는 무한 크롤링 | 중 (3개 파일, ~80줄) |
| **Phase 3** | **Author-Author 연결 정상화** + 그래프 품질 | 중 (2개 파일, ~100줄) |
| **Phase 4** | 대규모 데이터 대비 | 중 (3개 파일, ~80줄) |
| **Phase 5** | 사용자 유연성 | 필요시 결정 |

## 주의 사항

1. **Phase 1 적용 후 크롤링 속도 체크**: 필터 제거로 task 생성량이 급증할 수 있음.
   Phase 2의 적응형 딜레이와 함께 적용하는 것을 권장.

2. **DB 마이그레이션**: Phase 2의 retry_count/max_retries 추가 시 기존 DB 호환성 확인.
   `Column(..., default=0)` + `server_default="0"`으로 기존 레코드 호환.

3. **모니터링**: Phase 1 이후 depth 분포, task 생성률, API 호출률을 관찰하여
   적응형 딜레이의 threshold 값을 튜닝.

4. **롤백**: 각 Task는 독립적이므로 문제 발생 시 개별 롤백 가능.
